      \end{enumerate}
      \vspace{0.3em}
      {\small ONR Contract N00014-22-C-1011}
    \end{column}
    \begin{column}{0.42\textwidth}
      \textbf{Claims:}
      \begin{itemize}
        \item Outperform Sobel, Prewitt, Canny
        \item Outperform DexiNed, TEED, EDTER
        \item $<$1° angular accuracy
        \item Superior in low-SNR aquatic scenes
        \item ML is ``unreliable and dangerous''
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{How the WVF Works}
  \begin{columns}[T]
    \begin{column}{0.45\textwidth}
      2D Taylor expansion at each pixel:
      \begin{equation*}
        f(x_i, y_i) \approx f^0 + f_x^0 x + f_y^0 y + \frac{f_{xx}^0 x^2}{2} + \cdots
      \end{equation*}
      \begin{itemize}
        \item $N_p$ neighbors in circular region
        \item Least-squares solve per orientation
        \item Line Filter chains WVF along a line
        \item Papers use $N_p = 250$, $m = 14$
      \end{itemize}
    \end{column}
    \begin{column}{0.52\textwidth}
      \begin{tabular}{lcc}
        \toprule
        & \textbf{Sobel} & \textbf{WVF} \\
        \midrule
        Pixels used & 9 & 250 \\
        Kernel size & 3$\times$3 & $\sim$18px radius \\
        Orientations & 2 (x, y) & $N_s$ (any) \\
        Runtime & $\mu$s & seconds \\
        \bottomrule
      \end{tabular}
      \vspace{0.5em}

      {\small \textbf{28$\times$ more data} per pixel---but only compared against 3$\times$3.}
    \end{column}
  \end{columns}
\end{frame}

\sectionsubtitle{}
\section{Assessment}

\begin{frame}{Claims vs Evidence}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\small \color{red}$\times$} ``ML is unreliable and dangerous''\\
      {\footnotesize Cites only 2010/2016 sources.}\\[0.7em]
      {\small \color{red}$\times$} ``Divert efforts'' to image processing\\
      {\footnotesize Scope is aquatic edge detection only.}\\[0.7em]
      {\small \color{orange}$\sim$} Outperforms SOTA ML\\
      {\footnotesize True on $n=4$ dataset. No statistical power.}\\[0.7em]
      {\small \color{green!60!black}\checkmark} $<$1° angular accuracy\\
      {\footnotesize Valid on synthetic data. No error bounds.}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Statistical issues:}
      \begin{itemize}
        \item 4-image dataset, hand-labeled
        \item No confidence intervals
        \item DexiNed/TEED trained on BIPED test set
        \item No Gaussian-smoothed Sobel baseline
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Strengths \& Weaknesses}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\large \color{green!60!black}\textbf{Strengths}}
      \begin{enumerate}
        \item Orientation-specific gradients --- \textbf{novel}
        \item Cubic spline angles --- \textbf{elegant}
        \item Low-SNR performance --- \textbf{real gap}
        \item WVF + LF complementary
        \item Multi-scale GMM extension
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\large \color{red}\textbf{Weaknesses}}
      \begin{enumerate}
        \item Only 3$\times$3 baselines --- \textbf{unfair}
        \item No runtime analysis
        \item $n = 4$ dataset --- \textbf{no power}
        \item 10,000$\times$ slower
        \item ML claims unsupported
        \item Text overlap across papers
      \end{enumerate}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Recommendations}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{For the authors:}
      \begin{enumerate}
        \item Test Sobel at 9$\times$9, 15$\times$15
        \item GPU implementation + timing
        \item $\geq$50 images with independent annotators
        \item Confidence intervals on metrics
        \item Focus claims on angular accuracy
      \end{enumerate}
    \end{column}
    \begin{column}{0.46\textwidth}
      \textbf{The WVF's real niche:}\\[0.3em]
      Larger Sobel matches ODS/OIS, but \textbf{cannot} provide:
      \begin{itemize}
        \item Sub-degree angular precision
        \item Orientation-selective noise rejection
        \item Wave direction estimation
      \end{itemize}
      \vspace{0.3em}
      \textbf{Focus the narrative here.}
    \end{column}
  \end{columns}
\end{frame}
