\sectionsubtitle{What are the WVF and LF, and why do they matter?}
\section{Background}

\begin{frame}{The Papers Under Review}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Three publications} from USC Mechanical Engineering:
      \begin{enumerate}
        \item \textbf{M.S.\ Thesis} (2023)\\
              {\small Wide View \& Line Filter for Enhanced Image Gradient Computation and Edge Determination}
        \item \textbf{IEEE OCEANS 2024}\\
              {\small WVF \& LF for Enhanced Gradient Computation in Aquatic Environment}
        \item \textbf{IEEE OCEANS 2025}\\
              {\small Multi-Scale and Multi-Domain Edge Determination with Accurate Gradient Orientation}
      \end{enumerate}
      \vspace{0.5em}
      {\small Funded by ONR Contract N00014-22-C-1011}
    \end{column}
    \begin{column}{0.42\textwidth}
      \textbf{Core Claims:}
      \begin{itemize}
        \item WVF/LF outperform Sobel, Prewitt, Canny
        \item Outperform DexiNed, TEED, EDTER, BDCN
        \item $<$1° angular accuracy via cubic splines
        \item Superior in low-SNR aquatic environments
        \item ML is ``unreliable and dangerous''
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{How the WVF Works}
  \begin{columns}[T]
    \begin{column}{0.45\textwidth}
      \textbf{Key idea:} 2D Taylor expansion at each pixel
      \begin{equation*}
        f(x_i, y_i) \approx f^0 + f_x^0 x + f_y^0 y + \frac{f_{xx}^0 x^2}{2} + \cdots
      \end{equation*}

      \begin{itemize}
        \item $N_p$ neighbor pixels in circular region
        \item Solve $\mathbf{z} = (\mathbf{A}^T\mathbf{A})^{-1}\mathbf{A}^T\mathbf{b}$
        \item Rotated to $N_s$ orientations
        \item Line Filter chains WVF along a line
        \item Papers use $N_p = 250$, $m = 14$
      \end{itemize}
    \end{column}
    \begin{column}{0.52\textwidth}
      \textbf{WVF vs Traditional:}\\[0.5em]
      \begin{tabular}{lcc}
        \toprule
        & \textbf{Sobel} & \textbf{WVF} \\
        \midrule
        Pixels used & 9 & 250 \\
        Kernel size & 3$\times$3 & $\sim$18px radius \\
        Orientations & 2 (x, y) & $N_s$ (any) \\
        Adaptable & No & Yes \\
        Runtime & $\mu$s & seconds \\
        \bottomrule
      \end{tabular}
      \vspace{0.5em}

      {\small The WVF uses \textbf{28$\times$ more data} per pixel than Sobel---but the papers only compare against 3$\times$3 operators.}
    \end{column}
  \end{columns}
\end{frame}

\sectionsubtitle{What's wrong and what's right}
\section{Claims vs Evidence}

\begin{frame}{Overreaching Claims}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Claim 1:} ``ML is unreliable and dangerous'' for autonomous vehicles\\[0.5em]
      {\small \color{red}$\times$} Cites only 2010 and 2016 sources. Ignores massive advances in ML safety and robustness.\\[1em]

      \textbf{Claim 2:} Industry should ``divert efforts'' to image processing\\[0.5em]
      {\small \color{red}$\times$} Scope is edge detection in synthetic/aquatic images. Cannot generalize to all autonomous driving.
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Claim 3:} WVF outperforms SOTA ML models\\[0.5em]
      {\small \color{orange}$\sim$} True on their 4-image aquatic dataset, but $n=4$ has no statistical power. No confidence intervals reported.\\[1em]

      \textbf{Claim 4:} $<$1° angular accuracy\\[0.5em]
      {\small \color{green!60!black}\checkmark} Empirically demonstrated on synthetic data. Valid but lacks theoretical error bounds.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Statistical Concerns}
  \begin{columns}[T]
    \begin{column}{0.45\textwidth}
      \textbf{The 4-image aquatic dataset:}
      \begin{itemize}
        \item Hand-labeled by the authors
        \item No inter-annotator agreement
        \item No confidence intervals
        \item No $p$-values or effect sizes
        \item ODS/OIS differences of 2--5\% could be noise
      \end{itemize}

      \vspace{0.5em}
      \textbf{Standard practice:}\\
      BSDS500 uses 500 images with multiple annotators per image.
    \end{column}
    \begin{column}{0.52\textwidth}
      \textbf{BIPED test set concern:}\\[0.5em]
      DexiNed and TEED were \textit{trained} on BIPED.
      Testing them on BIPED (even test split) gives them a distribution advantage.\\[0.5em]

      The WVF/LF results on BIPED are therefore \textit{less} impressive, not more.\\[1em]

      \textbf{UDED results} are more informative (out-of-distribution), but margins are only 1--3\%.\\[0.5em]

      \textbf{Missing baselines:}
      \begin{itemize}
        \item No Gaussian-smoothed Sobel
        \item No LoG at matched scale
        \item No structured edge detectors
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\sectionsubtitle{Fair assessment of the contributions}
\section{Overall Assessment}

\begin{frame}{Publication Quality Scorecard}
  \begin{center}
    \includegraphics[width=0.82\textwidth]{figures/summary_scorecard.pdf}
  \end{center}
  \vspace{-0.5em}
  {\small \textbf{Math is excellent} (10/10). \textbf{Angular accuracy} and \textbf{low-SNR performance} are genuine strengths. \textbf{Experimental fairness}, \textbf{runtime analysis}, and \textbf{statistical validity} are the major weaknesses.}
\end{frame}

\begin{frame}{Strengths \& Weaknesses}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\large \color{green!60!black}\textbf{Strengths}}
      \begin{enumerate}
        \item Orientation-specific gradients via rotated Taylor expansion---\textbf{genuinely novel}
        \item Cubic spline angle detection---\textbf{elegant}, demonstrably superior to arctan
        \item Low-SNR aquatic performance fills a \textbf{real gap}
        \item WVF + LF are complementary (point vs line edges)
        \item Multi-scale GMM framework is a meaningful extension
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\large \color{red}\textbf{Weaknesses}}
      \begin{enumerate}
        \item Only compared to 3$\times$3 Sobel/Prewitt---\textbf{unfair}
        \item No runtime analysis---\textbf{critical omission}
        \item $n = 4$ aquatic dataset---\textbf{no statistical power}
        \item WVF is 10,000$\times$ slower---\textbf{impractical}
        \item Claims about ML ``unreliable and dangerous''---\textbf{unsupported}
        \item Significant text overlap between the three publications
      \end{enumerate}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Recommendations}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{For the authors:}
      \begin{enumerate}
        \item Compare against Sobel at matched spatial scale (9$\times$9, 15$\times$15)
        \item Publish GPU-optimized implementation with timing benchmarks
        \item Expand aquatic dataset to $\geq$50 images with independent annotators
        \item Report confidence intervals on all metrics
        \item Narrow claims to angular accuracy advantage---the genuine contribution
        \item Remove unsubstantiated claims about ML safety
      \end{enumerate}
    \end{column}
    \begin{column}{0.42\textwidth}
      \textbf{The WVF's real niche:}\\[0.5em]
      Orientation-specific gradient computation for angular accuracy in noise.\\[0.5em]

      Larger Sobel kernels can match ODS/OIS, but they \textbf{cannot} provide:
      \begin{itemize}
        \item Sub-degree angular precision
        \item Orientation-selective noise rejection
        \item Wave direction estimation
      \end{itemize}

      \vspace{0.5em}
      \textbf{Focus the narrative here.}
    \end{column}
  \end{columns}
\end{frame}
